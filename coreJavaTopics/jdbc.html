package com.database;

public class Employee {
	
private int eid;
private String ename;
private String desig;

public void setEid(int eid) {
	this.eid = eid;
}
public void setEname(String ename) {
	this.ename = ename;
}
public void setDesig(String desig) {
	this.desig = desig;
}

public int getEid() {
	return eid;
}
public String getEname() {
return ename;
}
public String getDesig() {
	return desig;
}
@Override
public String toString() {
	return " eid=" + eid + ",\n ename=" + ename + ",\n desig=" + desig;
}
}

package com.database;

import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.SQLException;

public class DAO {

public static Connection getDbConnect() {
	
try {
	Class.forName("com.mysql.cj.jdbc.Driver");
} 
catch (ClassNotFoundException e) {
	
	e.printStackTrace();
}
Connection conn=null;
try {
	conn= DriverManager.getConnection(
		"jdbc:mysql://localhost:3306/zdatabase", "root", "root"
	);
} 
catch (SQLException e) {

	e.printStackTrace();
}
		
return conn;
}

}

package com.database;

import java.sql.Connection;
import java.sql.PreparedStatement;
import java.sql.SQLException;

public class Controller {

public static void main(String[] args) throws SQLException {

Employee emp1  = new Employee(103,"Raju","Lead Developer");

//emp1.setEid(102);
//emp1.setEname("Jerry");
//emp1.setDesig("Tester");

System.out.println(emp1);

Connection conn = DAO.getDbConnect();

if(conn != null)
{
	System.out.println("DB Connected");
	
	String ins_query = "insert into employee1 values(?,?,?)";
	PreparedStatement stmt = conn.prepareStatement(ins_query);
	stmt.setInt(1, emp1.getEid());
	stmt.setString(2, emp1.getEname());
	stmt.setString(3, emp1.getDesig());
	int i = stmt.executeUpdate();
	if(i>0)
	{
		System.out.println("Inserted Successfully.");
	}
	else
	{
		System.out.println("Not Inserted.");
	}
}
else
{
	System.out.println("DB Not Connected");
}
		
}

}
