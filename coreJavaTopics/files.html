Writing data in Document using File 


package streams1;

import java.io.File;
import java.io.FileNotFoundException;
import java.io.IOException;
import java.io.PrintWriter;

public class datawriting 
{
  public static void main(String[] args) 
	{
		File f1 = new File("C:\\Users\\Admin\\Desktop\\datapro.txt");
		try {
			if(f1.createNewFile())
			{
				System.out.println(" file created "+f1.getName());
			}
			else
			{
				System.out.println(" file is existed ");
			}
		} catch (IOException e) {
	
			e.printStackTrace();
		}
	   
		try (PrintWriter output = new PrintWriter(f1)) {
			 output.print("\n data of the day.");
			 
			 output.print("\n\t User Name : Sri");
			 output.print("\n\t Purchased : Window Ac");
			 output.print("\n\t Price     : 49000/- Rs.");
			 
			 output.close();
		
		} catch (FileNotFoundException e) {
		
			e.printStackTrace();
		}
	}
}

Reading data in Document using File 


package streams1;

import java.io.File;
import java.io.FileNotFoundException;
import java.io.IOException;
import java.io.PrintWriter;

public class datawriting {

	public static void main(String[] args) {
	
		System.out.println("User Data Reading: ");
		File file = new File("C:\\Users\\Admin\\Desktop\\prodata101.txt");	
		
		try (Scanner input = new Scanner(file)) {
			
			while(input.hasNext())
			{
				String temp = input.next();
				
				System.out.print(temp+" ");
			}
		} 
		catch (FileNotFoundException e) {
			
			e.printStackTrace();
		}
	}

}
