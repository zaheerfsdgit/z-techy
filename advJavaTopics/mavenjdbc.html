  <version>0.0.1-SNAPSHOT</version>

  <properties>
    <project.build.sourceEncoding>UTF-8</project.build.sourceEncoding>

App.java / Controllers 

 
package Project1.SbiOnline;

import java.sql.Connection;
import java.sql.PreparedStatement;
import java.sql.SQLException;

public class App 
{
    public static void main( String[] args ) throws SQLException
    {
    System.out.println( "Hello World! From Maven" );
        
    Connection conn = DAO.getDbConnect();
    
    if(conn != null ) System.out.println("Db connected");
    else System.out.println("Db not ");
    
    PreparedStatement ps = conn.prepareStatement("Insert into employee values(103,'Ken','Project Manager')");
    
    int row = ps.executeUpdate();
    
    if(row>0) System.out.println("row inserted");
    else System.out.println("row not");
    }
}

DAO Class 

 
package Project1.SbiOnline;

import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.SQLException;

public class DAO {
	
public static Connection getDbConnect() {
		
    try {
        Class.forName("com.mysql.cj.jdbc.Driver");
    } 
    catch (ClassNotFoundException e) {
        
        e.printStackTrace();
    }
    Connection conn=null;
    try {
        conn= DriverManager.getConnection(
            "jdbc:mysql://localhost:3306/zdatabase", "root", "root"
        );
    } 
    catch (SQLException e) {
    
        e.printStackTrace();
    }
    
    return conn;
	}

}
